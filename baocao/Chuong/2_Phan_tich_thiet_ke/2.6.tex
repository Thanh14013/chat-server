\section{Hệ thống Quản trị và Chống lạm dụng}

\subsection{Kiểm soát Rủi ro Tấn công Từ chối Dịch vụ (DoS/DDoS)}

\begin{enumerate}
    \item Cơ sở lý thuyết về Kiểm soát Tấn công Từ chối Dịch vụ (DoS/DDoS)
    
    Các máy chủ TCP khi mở cổng giao tiếp công khai luôn tiềm ẩn rủi ro quá tải do lượng lớn yêu cầu mạng (Flooding). Cụ thể, các hình thức tấn công Từ chối Dịch vụ Phân tán (DDoS) tại Tầng 7 (Application Layer) thường khai thác các thông điệp hợp lệ nhằm làm cạn kiệt tài nguyên xử lý (CPU) và bộ nhớ (RAM).
    
    Dựa trên nguyên lý Bảo mật Chiều sâu (Defense in Depth), hệ thống tích hợp cơ chế kiểm soát truy cập thông qua thuật toán Giỏ thẻ bài (Token Bucket). Mô hình này cấp phát một số lượng thẻ (Tokens) nhất định trên mỗi kết nối, trong đó mỗi tác vụ tiêu thụ một lượng thẻ tương ứng. Khối lượng thẻ được tái nạp theo tốc độ cố định (Refill Rate). Kết hợp cùng Hệ thống Phát hiện Xâm nhập (IDS), hành vi của máy khách được phân tích và đánh giá qua Điểm rủi ro (Threat Scoring) nhằm thiết lập cơ chế từ chối dịch vụ vô thời hạn (Permanent Ban) đối với các tác nhân vi phạm.

    \item Triển khai Hệ thống RateLimiter và IntrusionDetector
    
    Hệ thống kiểm soát được cấu thành từ hai phân hệ độc lập: RateLimiter và IntrusionDetector:
    
    \begin{itemize}[label={-}]
        \item Tầng 1 - Bộ giới hạn tốc độ (RateLimiter - Token Bucket): Hàm \texttt{TokenBucket::}\allowbreak\texttt{consume()} liên tục trừ điểm Token dựa trên tham số thời gian trễ (\texttt{elapsed * refill\_rate}). Các ngưỡng giới hạn được áp dụng phụ thuộc vào loại tác vụ (\texttt{MSG\_CHAT}, \texttt{CONNECT}, \texttt{AUTH}). Trong trường hợp máy khách tiêu thụ hết Token và số lần vi phạm vượt mức cho phép (\texttt{violations >= 4}), các yêu cầu tiếp theo sẽ bị từ chối (\texttt{RATE\_LIMITED}) nhằm giảm tải CPU.
        
        \item Tầng 2 - Hệ thống chấm điểm Rủi ro (IntrusionDetector): Các vi phạm ghi nhận từ \texttt{RateLimiter} và các phân hệ khác được chuyển giao qua hàm \texttt{reportViolation()}. Mỗi mã lỗi được quy đổi thành điểm số rủi ro (\texttt{Threat Score}) tương ứng với mức độ nghiêm trọng.
        
        \item Tầng 3 - Biện pháp Xử lý (Progressive Penalty): Hàm \texttt{checkThresholds()} tiến hành phân tích tổng điểm và thực thi biện pháp kiểm soát:
        \begin{itemize}[label={$\circ$}]
            \item Ngưỡng 100 điểm (\texttt{THREAT\_THRESHOLD}): Đưa IP vào danh sách hạn chế 1 giờ (\texttt{m\_tempBans}).
            \item Ngưỡng 200 điểm (\texttt{ATTACK\_THRESHOLD}): Đưa IP vào danh sách hạn chế 24 giờ.
            \item Ngưỡng 500 điểm (\texttt{PERM\_BAN\_THRESHOLD}): Lưu vết sự kiện (\texttt{AuditLogger}), xuất danh sách (\texttt{exportBanList()}) và bổ sung IP vào \texttt{m\_permBans} nhằm từ chối dịch vụ vô thời hạn ở cấp độ mạng.
        \end{itemize}
    \end{itemize}
\end{enumerate}

\subsection{Kiểm soát Truy cập và Quản trị (RBAC)}

\begin{enumerate}
    \item Cơ sở lý thuyết về Kiểm soát Truy cập dựa trên Vai trò (RBAC)
    
    Mô hình Kiểm soát Truy cập dựa trên Vai trò (Role-Based Access Control - RBAC) được triển khai nhằm quản trị đặc quyền hệ thống. Theo mô hình này, người dùng được phân bổ vào các Vai trò (Roles) khả dụng thay vì được cấp phát quyền trực tiếp. Hệ thống tuân thủ Nguyên tắc Đặc quyền tối thiểu (Principle of Least Privilege), cấu hình tài khoản mới ở mức quyền hạn cơ bản nhất. Các tác vụ can thiệp vào trạng thái quản trị đều được lưu vết vào Nhật ký Kiểm toán (Audit Trails) nhằm duy trì Tính Trách nhiệm giải trình (Accountability).

    \item Triển khai Phân quyền trong mã nguồn (AdminCommands)
    
    Hệ thống kiến trúc 3 cấp bậc quyền hạn (\texttt{USER}, \texttt{ADMIN}, \texttt{OWNER}), được điều phối bởi lớp \texttt{AdminCommands}. Tiến trình kiểm tra đặc quyền được vận hành tập trung qua hàm \texttt{hasAdminRight(adminFd, ownerOnly)} nhằm giới hạn rủi ro leo thang đặc quyền.
    
    \begin{itemize}[label={-}]
        \item \texttt{USER} (Mặc định): Được cấp phát quyền giao tiếp cơ bản (Nhắn tin, Truyền tệp), chịu sự giám sát toàn diện của \texttt{RateLimiter} và \texttt{IntrusionDetector}.
        
        \item \texttt{ADMIN} (Quản trị viên): Được cấp phát đặc quyền vận hành, cho phép thực thi các lệnh điều khiển:
        \begin{itemize}[label={$\circ$}]
            \item \texttt{/mute} \& \texttt{/unmute}: Giới hạn hoặc khôi phục tính năng gửi tin nhắn của một phiên.
            \item \texttt{/kick} \& \texttt{/unkick}: Hủy tư cách thành viên phòng thông qua \texttt{kickFromRoom()}.
            \item \texttt{/ban} \& \texttt{/unban}: Tương tác với \texttt{IntrusionDetector::permBan()} nhằm khóa vô thời hạn địa chỉ IP và định danh.
            \item \texttt{/broadcast}: Phân phối thông điệp hệ thống tới toàn bộ phòng trò chuyện.
        \end{itemize}
        
        \item \texttt{OWNER} (Chủ sở hữu): Được cấp quyền quản trị tối cao (\texttt{ownerOnly = true}), bao gồm năng lực thực thi lệnh \texttt{/promote} (nâng cấp \texttt{USER} thành \texttt{ADMIN}) và \texttt{/demote} (giáng cấp \texttt{ADMIN} thành \texttt{USER}).
        
        \item Kiểm toán Đặc quyền (Audit Logging): Toàn bộ lệnh quản trị được thực thi thành công sẽ kích hoạt macro \texttt{AUDIT\_D()} nhằm lưu trữ bản ghi qua \texttt{AuditLogger}. Đối với lệnh \texttt{/kick}, hệ thống lưu vết \texttt{AuditEventType::ADMIN}, bao gồm định danh các bên liên quan và tham số lý do (\texttt{reason}) nhằm đáp ứng quy chuẩn thanh tra bảo mật.
    \end{itemize}
\end{enumerate}
